a.combine(b).combine(c) == a.combine(b.combine(c))

* Identity

\[\begin{aligned} a \circ identity_{a} == a == identity_{a} \circ a \\ \end{aligned} \] 

a.combine(empty()) == a == empty<A>().combine(a)

Monoid - Definition


                interface Monoid<F> {

                  fun F.combine(b: F): F

                  fun empty(): F

                  operator fun F.plus(b: F): F = this.combine(b)
                  // monoidA.combine(monoidB) → monoidA + monoidB
                }

Monoid - Implementation


                object StringMonoid : Monoid<String> {

                  override fun String.combine(b: String): String = "${this}$b"

                  override fun empty(): String = ""
                }
            


                object IntMonoid : Monoid<Int> {

                  override fun Int.combine(b: Int): Int = this + b

                  override fun empty(): Int = 0
                }

Monoid - Testing


                fun <F> Monoid<F>.associativityLaw(a: F, b: F, c: F) =
                  Assert.assertEquals(
                    a.combine(b).combine(c),
                    a.combine(b.combine(c))
                  )
            


                fun <F> Monoid<F>.identityLaw(a: F) {
                  Assert.assertEquals(a, empty().combine(a))
                  Assert.assertEquals(a, a.combine(empty()))
                }
