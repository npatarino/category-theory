                sealed class Option<out F> : Functor<F> {

                    data class Some<F>(val value: F) : Option<F>()

                    object None : Option<Nothing>()

                    override fun <B> map(transformation: (F) -> B): Option<B> = when (this) {
                        is Some -> Some(transformation(value))
                        None -> None
                    }
